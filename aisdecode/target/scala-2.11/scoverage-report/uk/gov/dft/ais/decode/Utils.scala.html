12   /**
13    * The data payload is an ASCII-encoded bit vector.
14    * Each character represents six bits of data.
15    * To recover the six bits, subtract 48 from the ASCII character
16    * value; if the result is greater than 40 subtract 8.
17    * According to [IEC-PAS], the valid ASCII characters for
18    * this encoding begin with "0" (64) and end with "w" (87);
19    * however, the intermediate range "X" (88) to "\_" (95)
20    * is not used.
21    * [encodedMessage ASCII encoded bit vector]
22    */
23   def ais_to_binary(encodedMessage:String): String = {
24     encodedMessage
25       // This step implements the above
26       .map(ch => ch.toInt)
27       .map(e => e - 48)
28       .map {case e if e > 40 => e - 8; case e => e}
29       // This step converts to a long binary string
30       .map(e => String
31         .format("%6s", Integer.toBinaryString(e))
32         .replace(' ', '0')
33       ).mkString
34   }

36     /**
37      * To decode text encoded in AIS we need to convert
38      * Six bit ASCII into regular ASCI
39      * see: http://www.catb.org/gpsd/AIVDM.html#_ais_payload_data_types
40      * [binarystring a string containing binary 6bit ASCII]
41      */
42     def ais_6bit_asci(binarystring:String): String = {
43     binarystring
44       // split the string into 6 bit chunks
45       .split("(?<=\\G.{6})")
46       // convert binary to Ints
47       .map(c => Integer.parseInt(c, 2))
48       // implement conversion of strange 6bit ASCII
49       .map {case i if i <= 31 => i + 64;
50             case i if i>31 => i;
51             case i => i}
52       // now we have regular ASCII integers convert to char
53       .map {i => i.toChar}.mkString
54   }

57   /**
58     * Calculate the XOR checksum for AIS arrays.
59     * @param stringChecksumPair An array of the form [String, Checksum]
60     * @return boolean True (checksum match) or False
61     */
62    def validate_ais_checksum(stringChecksumPair: Array[String]): Boolean =
63      stringChecksumPair match {
64        case Array(string, checksum_match) =>
65          var checksum = 0
66          // Map over the string performing XOR checksum calculation
67          string.foreach{  char => checksum = checksum ^ char.toInt }
68          if( f"$checksum%02X".toUpperCase == checksum_match.toUpperCase){
69            true //Checksum matches!
70          } else {
71            println(f"""Error in checksum.
72              Expected $checksum_match%h
73              Found $checksum%02x
74              For string: $string""")
75            false //Checksum doesn't match
76          }
77        case _ => println("Error! Not an array!"); false
78      }

81     /**
82      * Split and clean the checksums parts of the message then calculate result
83      * Only returns true if both pass.
84      * [message a string from the ais data file]
85      */
86     def process_checksum(message: String): Boolean = {
87       // Split the strings using the !
88       val validation_segments = message.split("!")
89       // clean the leading and training \ from the AIS message
90       // note that these are escpaed twice... \\\\ is \\ in regex
91       val cleaned_messages = validation_segments
92         .map { s => s.replaceAll("(^\\\\)|(\\\\$)", "")}
93       // split messages and checksums into array [message, checksum]
94       val split_messages = cleaned_messages.map{ pair => pair.split("\\*")}
95       // For each checksum in a message pair see if it passes
96       val result = split_messages.map{pair => validate_ais_checksum(pair)}
97       // Only return true if both parts of the message pass checksum test
98       result.forall(_ == true)
99     }

109      * Parse integers that have a scale (i.e. negative numbers too)
110      */
111     def parseIntWScale(s: String): Option[Double] = {
112       // parse the int as per usual
113       val pInt= parseIntSafe(s).getOrElse(0)
114 
115       // if the integer is larger than it should be given the length then it
116       // is a negative number (i.e. the scale)
117       if (pInt> pow (2, s.length-1) ) {
118         Some(pInt - pow(2, s.length))
119       }
120       // if not then it is a regular Int
121       else {
122         Some(pInt)
123       }
124     }
