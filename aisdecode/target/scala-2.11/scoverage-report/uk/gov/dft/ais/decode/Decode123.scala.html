78     val getROT = udf [Option[Double] , String] {x=>
79       // Values between 0 and 708 degrees/min coded by
80       // ROTAIS=4.733 * SQRT(ROTsensor) degrees/min
81       // where ROTsensor is the Rate of Turn as input by an external Rate of
82       // Turn Indicator. ROTAIS is rounded to the nearest integer value.
83       // Thus, to decode the field value, divide by 4.733 and then square it.
84       // Sign of the field value should be preserved when squaring it,
85       // otherwise the left/right indication will be lost.
86       val ROT = parseIntWScale(x.slice(42, 50)).getOrElse(0.0)
87       val ROTdeg = ROT / 4.733
88       val squaredROT = pow(ROTdeg,2)
89       // we lost the sign during squaring
90       if(ROT <0){Some(-squaredROT)}
91       // 128 actually means none
92       else if(abs(ROT)==128){None}
93       else{Some(squaredROT)}
94     }

96     val getSOG = udf [Option[Double] , String] {x=>
97       // Speed over ground is in 0.1-knot resolution from 0 to 102 knots.
98       // Value 1023 indicates speed is not available

100       extractInt(x,50,60) match {
101         // If we have some scale by factor, unless 1023, then none!
102         case Some(i) => if(i == 1023){None}else{Some(i / 10.0)}
